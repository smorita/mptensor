257 template <template <typename> class Matrix, typename C>
258 inline void Tensor<Matrix, C>::make_l2g_map() const {
259  const size_t rank = Dim.size();
260  const size_t rank0 = upper_rank;
261  const size_t rank1 = rank - rank0; // lower_rank
262  const size_t l_row = Mat.local_row_size();
263  const size_t l_col = Mat.local_col_size();
264  const size_t *axes_map0 = &(axes_map[0]);
265  const size_t *axes_map1 = &(axes_map[rank0]);
266 
267  l2g_map_row.resize(l_row * rank0);
268  l2g_map_col.resize(l_col * rank1);
269 
270 #pragma omp parallel default(shared)
271  {
272  // Type int instead of size_t is used because of std::div()
273  int dim0[rank0];
274  int dim1[rank1];
275  for (size_t i = 0; i < rank0; ++i) dim0[i] = int(Dim[axes_map0[i]]);
276  for (size_t i = 0; i < rank1; ++i) dim1[i] = int(Dim[axes_map1[i]]);
277 
278  int g_row, g_col; // global matrix row- and column-index
279  std::div_t divresult;
280  size_t *map;
281 #pragma omp for
282  for (size_t k = 0; k < l_row; ++k) {
283  map = &(l2g_map_row[k * rank0]);
284  g_row = int(Mat.global_row_index(k));
285  for (size_t i = 0; i < rank0; ++i) {
286  divresult = std::div(g_row, dim0[i]);
287  map[i] = divresult.rem;
288  g_row = divresult.quot;
289  }
290  }
291 #pragma omp for
292  for (size_t k = 0; k < l_col; ++k) {
293  map = &(l2g_map_col[k * rank1]);
294  g_col = int(Mat.global_col_index(k));
295  for (size_t i = 0; i < rank1; ++i) {
296  divresult = std::div(g_col, dim1[i]);
297  map[i] = divresult.rem;
298  g_col = divresult.quot;
299  }
300  }
301  }
302 
303  return;
304 };

604 template <template <typename> class Matrix, typename C>
605 void Tensor<Matrix, C>::change_configuration(const size_t new_upper_rank,
606  const Axes &new_axes_map) {
607  if ((upper_rank == new_upper_rank) && (axes_map == new_axes_map)) return;
608 
609  Tensor<Matrix, C> T_old(*this);
610 
611  const size_t rank = this->rank();
612  Shape dim;
613  Axes axes;
614  dim.resize(rank);
615  axes.resize(rank);
616  for (size_t i = 0; i < rank; ++i) {
617  dim[i] = Dim[new_axes_map[i]];
618  axes[new_axes_map[i]] = i;
619  }
620  init(dim, new_upper_rank);
621  transpose(axes);
622 
623  /* create lists of local position and destination rank */
624  const size_t local_size = T_old.local_size();
625  std::vector<int> dest_mpirank(local_size);
626  std::vector<unsigned long int> local_position(local_size);
627 
628  T_old.prep_local_to_global();
629  this->prep_global_to_local();
630 
631 #pragma omp parallel default(shared)
632  {
633  Index index;
634  index.resize(rank);
635 
636 #pragma omp for
637  for (size_t i = 0; i < local_size; ++i) {
638  T_old.global_index_fast(i, index);
639 
640  int dest;
641  size_t pos;
642  this->local_position(index, dest, pos);
643 
644  local_position[i] = pos;
645  dest_mpirank[i] = dest;
646  }
647  }
648 
649  /* exchange data */
650  replace_matrix_data(T_old.get_matrix(), dest_mpirank, local_position, Mat);
651 
652  return;

1058  }

1063  dim_new.resize(rank);
1064  for (size_t r = 0; r < rank; ++r) {
1065  dim_new[r] = dim_old[axes[r]];
1066  }
1067

1070

1072  const size_t local_size = T.local_size();
1073  std::vector<int> dest_mpirank(local_size);
1074  std::vector<unsigned long int> local_position(local_size);
1075 
1076  T.make_l2g_map();
1077  T_new.prep_global_to_local();
1078 
1079 #pragma omp parallel default(shared)
1080  {
1081  // Assuming axes_map of T_new is [0,1,2,...]
1082  size_t d_offset[rank];

1084  for (size_t r = 0; r < urank_new; ++r) {

1087  }
1088  for (size_t r = urank_new; r < rank; ++r) {

1090  d_col *= dim_new[r];

1098  }
1099  for (size_t r = 0; r < rank; ++r) {
1100  axes_trans[r] = axes_inv[axes_map[r]];
1101  }
1102 
1103  size_t index_new[rank];

1105 #pragma omp for
1106  for (size_t i = 0; i < local_size; ++i) {
1107  T.global_index_l2g_map_transpose(i, axes_trans, index_new);
1108

1110  g_col(0); // indices of global matrix for transposed tensor

1112  g_row += index_new[r] * d_offset[r];

1114  for (size_t r = urank_new; r < rank; ++r) {
1115  g_col += index_new[r] * d_offset[r];
1116  }
1117 
1118  T_new.local_position_fast(g_row, g_col, dest_mpirank[i],
1119  local_position[i]);
1120  }
1121  }
1122 
1123  /* exchange data */
1124  replace_matrix_data(T.get_matrix(), dest_mpirank, local_position,
1125  T_new.get_matrix());
1126 
1127  return T_new;
